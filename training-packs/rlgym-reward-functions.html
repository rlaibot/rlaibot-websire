RLGym Reward Functions

RLGym Reward Functions Machine Learning Python Advanced

Description

A collection of reward functions for RLGym that can be used to train RL agents. This example demonstrates how to implement custom rewards for RL training, including velocity-based rewards, alignment rewards, and touch rewards.

Code

# Example RLGym Reward Function
from rlgym.utils.reward_functions import RewardFunction
from rlgym.utils.gamestates import GameState, PlayerData
import numpy as np

class CustomReward(RewardFunction):
    def __init__(self):
        super().__init__()
        # Initialize any parameters here
        self.velocity_weight = 0.1
        self.alignment_weight = 0.3
        
    def reset(self, initial_state: GameState):
        # Reset any state between episodes
        pass
    
    def get_reward(self, player: PlayerData, state: GameState, 
                  previous_action: np.ndarray) -> float:
        # Calculate reward components
        ball = state.ball
        car = player.car_data
        
        # 1. Velocity reward (encourage moving towards ball)
        velocity_reward = np.dot(car.linear_velocity, 
                               ball.position - car.position) * self.velocity_weight
        
        # 2. Alignment reward (face towards ball)
        to_ball = ball.position - car.position
        to_ball_norm = to_ball / np.linalg.norm(to_ball)
        alignment = np.dot(car.forward(), to_ball_norm)
        alignment_reward = alignment * self.alignment_weight
        
        # 3. Touch reward (if ball was touched)
        touch_reward = 1.0 if player.ball_touched else 0.0
        
        # Combine rewards
        total_reward = velocity_reward + alignment_reward + touch_reward
        return float(total_reward)

# Usage example:
from rlgym.utils.reward_functions import CombinedReward
from rlgym_tools.extra_rewards import EventReward

def get_reward():
    return CombinedReward(
        (CustomReward(), 1.0),
        (EventReward(goal=10, concede=-10), 1.0)
    )
